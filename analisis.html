📝 Código del Ejemplo


"""
Crear una frase larga con palabras repetias donde exprese un sentimiento positivo
y sentimientos negativos, analizar la frase cimpleta, eliminando signos de puntuacion,
tokenizando la frase, y recorriendo la frase con for, con if contabilizar cuántas
palabras positivas y cuantas negativas s encontraron, y de ahi determinar
si la frase es positiva o negativa
"""
#funciones clave: replace, split, lower, for, if
#creando listas con las palabras  para detectar el sentir de la frase
palabraspositivas=["feliz","contento","alegre","maravilloso","maravillosa","bonito","bonita"]
palabrasnegativas=["triste","solo","aburrido","malo","deprimido","horrible","muerte"]

positivo=0
negativo=0

frase=input("Ingresa tu frase: ")

frase_limpio=frase.replace(",","").replace(".","").replace(":","").lower()
tokens = frase_limpio.lower().split()

verificadas=[]

for palabra in tokens:
    if palabra not in verificadas:
        verificadas.append(palabra)

        if palabra in palabraspositivas:
            positivo += 1
        elif palabra in palabrasnegativas:
            negativo += 1

print("la cantidad de palabras positivas son: ",positivo)
print("la cantidad de palabras negativas son: ",negativo)

if positivo>negativo:
    print("La frase es positivo")
elif negativo>positivo:
    print("La frase es negativo")
else:
    print("Su frase no se orinta a ningun sentimiento en específico")
#me hizo sentir feliz y muy contento que tuve una mascota muy bonita, maravillosa y alegre

replace

para reemplazar todas las ocurrencias de un patrón en una cadena
