from deep_translator import GoogleTranslator

texto = input("Ingresa la palabra o frase para traducir: ")

    print("=== CONTROL DE INVENTARIO ===")
    print("1- [e] Espanol")
    print("3- [i] Ingles")
    print("4- [f] Frances")
    print("5- [a] Aleman")
    print("6- [y] Italiano")
    print("7- [p] Portugues")
    print("8- [x] Salir del programa")
    opc=input("Selecciona una opcion: ")

    if opc=="e":
        espanol()
    elif opc=="i":
        ingles()
    elif opc=="f":
        frances()
    elif opc=="a":
        aleman()
    elif opc=="y":
        italiano()
    elif opc=="p":
        portugues()

        print("Saliendo del programa...")
